This is a Simple Component to List ToDo List in https://jsonplaceholder.typicode.com/todos

## Usage
```javascript
import React, { Component } from 'react';
import { Todo } from 'documint-ui';

class MyApp extends Component {
  render() {
    return (
      <div>
        <Todo/>
      </div>
    );
  }
}

export default MyApp;
```
